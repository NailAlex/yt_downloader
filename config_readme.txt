Описание параметров конфига

  "test_mode" - true/false 				       		//Включение выключение диагностической телеметрии
  "test_telebot" - true/false						//Включение выключение доп. диагностической телеметрии	в процессе телебота
  "test_dump_json" - true/false					//Включение выключение дампа в json данных о медиафайле
  "test_dump_ytdlp_cmd" - true/false				//Включение выключение дампа в команд к yt-dlp
  "name" - "YT-Downloader_Bot",					//Имя бота(заглушка на будущее)
  "t_name" - "YTD_Telebot"						//Имя процесса(заглушка на будущее)	
  "sleep_time" - 2								//Время цикла опроса в секундах(доступны дробные значения типа 0.2)
  "ping_time" - 1								//Время внутреннего пинга процессов(живо или дохло, повисло-не повисло) в минутах.
  "fail_ping_count" - 3							//Число пропущенных пингов для процесса для активации его автоматического перезапуска.
  "ether_iface_name" - "eth0"						//для Linux, имя сетевого интерфейса железки. Его он будет использовать для получения данных для команды /info 
  "save_description" - true/false					//сохранять или нет описание видео в отдельном файле(.description)?
  "storage_auroclear_period_days" - 30,			//автоочистка хранилища(в разработке).
  "last_storage_auroclear_date" - "2023-12-22" 		//автоочистка хранилища(в разработке).
  "log_logfile" - "./log/yt_downloader.log"				//путь к основному логфайлу бота
  "log_errfile" - "./log/yt_downloader_error.log"		//путь к логфайлу ошибок бота
  "log_worker_logfile" - "./log/yt_downloader_worker.log" 			//путь к логфайлу воркеров-загрузчиков
  "log_worker_errfile" - "./log/yt_downloader_worker_error.log"		//путь к логфайлу ошибок воркеров-загрузчиков
  "url_cache_file" - "./data/cache/downloads.json"					//путь к файлу-списку текущих загрузок(отдельных URL)
  "jobs_cache_file" - "./data/cache/jobs.json"						//путь к файлу-списку текущих задач(от пользователей)
  "cookies_path" - "/data/cookies" 								//путь к папке для временных куки(под каждого воркера)
  "cookies_src_file" - "/src/yt_cookies.txt",						//путь к файлу-куки(мастер куки для дублирования под каждого воркера)
  "dumps_savepath" - "/data/dumps"							//путь к папке для отладочных дампов
  "local_savepath" - "/data/storage"  							//путь к папке для хранения скачанного медиа на ЛОКАЛЬНОМ хранилище железки где исполняется скрипт. Должен быть "" если используется удаленное хранилище на NAS.
  "remote_savepath" - "" 										//путь к папке для хранения скачанного медиа на УДАЛЕННОМ хранилище. Должен быть "" если используется локальное хранилище на железке.
  "max_workers" - 4											//количество паралельных воркеров загружающих медиа. Рекомендуется не более 4 если канал дохлый(100Мбит и менее).
  "max_worker_bandwidth" - "3M"								//ограничение скорости в мегабайтах/с для воркера. Не рекомендуется давать каждому полный канал. 
														//При 100Мбит общая скорость 4 воркеров заполнит весь канал при 3М.
  "TOKEN" - ""												//Токен бота
  "NAME" - "MyCooolYTDownloader_bot"						//Имя бота отображаемое в канале админа и в каналах работы с ботом
  "CHANID" - [111111111111] 								//Список каналов для контроля(заглушка на будущее, всегда прописывается канал самого админа бота)
  "CONTROL_CHANID" - 111111111111						//Номер канала админа бота, куда будет отправляться уведомления о перезапуске и куда можно получить расширенные данные(логи и проч)

Для получения номера канала, запустить бота, зайти к нему в личку по ссылке от BotFather, сделать /start и затем /info, в ответ он отправит данные по железке и ID канала.
